# app/modules/crm/services.py içine eklenecek/değiştirilecek

import json
import google.generativeai as genai
import os

# API Key'inizi buraya girin (veya .env dosyasından çekin)
genai.configure(api_key=os.environ.get("GEMINI_API_KEY", "BURAYA_API_ANAHTARINIZI_YAZIN"))

class CrmService:
    
    @staticmethod
    def _gercek_ai_sentiment(metin):
        """Gerçek LLM API kullanarak Müşteri Duygu Analizi Yapar"""
        if not metin: return 'BELIRSIZ', 5
        
        try:
            model = genai.GenerativeModel('gemini-1.5-flash')
            prompt = f"""
            Sen uzman bir müşteri ilişkileri (CRM) yapay zeka asistanısın. 
            Aşağıdaki müşteri görüşme notunu analiz et ve müşterinin duygu durumunu sınıflandır.
            Sadece aşağıdaki JSON formatında yanıt ver, başka hiçbir metin ekleme:
            {{"duygu": "MUTLU" veya "NORMAL" veya "MUTSUZ" veya "SINIRLI" veya "BELIRSIZ", "skor": 1-10 arası tam sayı}}
            
            Görüşme Notu: "{metin}"
            """
            
            response = model.generate_content(prompt)
            # LLM'den dönen markdown taglerini temizleyip JSON'a çeviriyoruz
            temiz_metin = response.text.replace('```json', '').replace('```', '').strip()
            sonuc = json.loads(temiz_metin)
            
            return sonuc.get('duygu', 'BELIRSIZ'), sonuc.get('skor', 5)
        except Exception as e:
            import logging
            logging.getLogger(__name__).error(f"LLM API Hatası: {str(e)}")
            return 'BELIRSIZ', 5
